## JS小练习

```js

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <meta http-equiv="X-UA-Compatible" content="ie=edge">
    <title>Document</title>

    <style>
        .cs1{
            color: red;
            font-size: 10px;
        }
        .cs2{
            color: pink;
            font-size: 20px;
        }
        .cs3{
            color: blue;
            font-size: 30px;
        }
    </style>

    <script>

        var i = 0;

        // 定义一个用来改变样式的函数
        function changeStyle(){
            // 获取标签
            var op = document.getElementById('pd');
            styleName = 'cs' + (i % 3 + 1);
            op.className = styleName;
            i++;
        }

        // js程序入口
        window.onload = function(){
            // 定动定时器
            t = setInterval(changeStyle,1000);

        }
    </script>


</head>
<body>
    <!-- 
        通过定时器来让一个标签的样式每隔3秒换一下
     -->

     <p id="pd">AAAAAAAA</p>
</body>
</html>
```
