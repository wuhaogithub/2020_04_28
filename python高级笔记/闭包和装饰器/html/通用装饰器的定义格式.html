通用装饰器的定义格式

格式:

def outer(func):
    def inner(*args, **kwargs):
        ret = func(*args, **kwargs)    
        return ret
    return inner

'''
P10_为不同的函数来定义装饰器
'''


# 下面这个格式就是通用装饰器的定义格式
def outer(func):
    def inner(*args, **kwargs):
        print('函数调用前的装饰功能代码...')
        ret = func(*args, **kwargs)
        print('函数调用后的装饰功能代码...')
        return ret

    return inner

@outer
def show1():
    print('Show 1 Run....')

@outer
def show2(msg):
    print('Show 2 Run ... ' + msg )

@outer
def show3():
    # print('Show 3 Run ...')
    return 'Show 3 Ret ...'


def show4(msg):
    print('Show 4 Run ...' + msg)
    return 'Show 4 Ret ...' + msg

@outer
def show5(msg,n):
    print('Show 5 Run ...' + msg , n)
    return 'Show 5 Ret ...' + msg + str(n)

show1()
show2('Hello')
ret = show3()
print('Show3返回值为:', ret)
ret = show4('World')
print('Show4返回值为:', ret)

ret = show5('World',1000)
print('Show5返回值为:', ret)
